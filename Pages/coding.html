                        \documentclass{standalone}
\usepackage{filecontents}
\usepackage{float}
\usepackage{pgffor}
\usepackage{pgfplots}
\pgfplotsset{compat=1.17}

% Define the Lotka-Volterra equations in the preamble
\newcommand{\fx}[2]{\alpha*#1 - \beta*#1*#2} % Prey equation: dx/dt = \alpha*x - \beta*x*y
\newcommand{\fy}[2]{\delta*#1*#2 - \gamma*#2} % Predator equation: dy/dt = \delta*x*y - \gamma*y

\newwrite\datfile

\AtBeginDocument{%
    % Parameters for the Lotka-Volterra model
    \def\alpha{0.1} % Prey birth rate
    \def\beta{0.02} % Predation rate
    \def\gamma{0.1} % Predator death rate
    \def\delta{0.01} % Predator reproduction rate
    
    % Initial conditions
    \def\xinitial{40} % Initial prey population
    \def\yinitial{9} % Initial predator population
    \def\h{0.1} % Step size
    \def\nsteps{5000} % Number of steps
    
    % Open the data.dat file for appending
    \immediate\openout\datfile=dataLV.dat
    
    % Write the initial values
    \def\x{\xinitial}
    \def\y{\yinitial}
    \immediate\write\datfile{0\space \x\space \y} % Write time, initial x, initial y
    
    % Perform Euler's method and store the values
    \foreach \i in {1, 2, ..., \nsteps} {%
        \pgfmathsetmacro{\xnext}{\x + \h * (\fx{\x}{\y})}
        \pgfmathsetmacro{\ynext}{\y + \h * (\fy{\x}{\y})}
        \pgfmathsetmacro{\t}{\i * \h}
        \immediate\write\datfile{\t\space \xnext\space \ynext}
        \xdef\x{\xnext}
        \xdef\y{\ynext}
    }
    
    % Close the data.dat file
    \immediate\closeout\datfile
}

\begin{document}
    
    % Plotting the prey and predator populations over time
        \begin{tikzpicture}
            \begin{axis}[
                title={Lotka-Volterra Model: Prey and Predator Populations over Time},
                xlabel={Time ($t$)},
                ylabel={Population},
                grid=both,
                width=12cm,
                height=8cm,
                legend pos=north east
                ]
                \addplot table [x index=0, y index=1, col sep=space] {dataLV.dat};
                \addplot table [x index=0, y index=2, col sep=space] {dataLV.dat};
                \legend{Prey ($x$), Predator ($y$)}
            \end{axis}
        \end{tikzpicture}
    
\end{document}
